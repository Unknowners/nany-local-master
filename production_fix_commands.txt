# Команди для виправлення продакшн сервера
# Копіювати та виконати на продакшн сервері:

# 1. Перевірити міграції
docker-compose exec backend alembic current
docker-compose exec backend alembic history

# 2. Застосувати міграції
docker-compose exec backend alembic upgrade head

# 3. Завантажити дані
docker-compose exec backend python scripts/seed_data_complete.py

# 4. Перевірити результат
curl 'https://nany.datavertex.me/api/v1/onboarding_configs' | jq '. | length'
curl 'https://nany.datavertex.me/api/v1/data_categories' | jq '. | length'

# Очікувані результати:
# - onboarding_configs: 2
# - data_categories: 14
